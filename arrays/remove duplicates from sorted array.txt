
static int removeDuplicates(int[] arr) {
        int n = arr.length;
        if (n <= 1)
            return n;
		
      	// Start from the second element
        int idx = 1; 
      
        for (int i = 1; i < n; i++) {
            if (arr[i] != arr[i - 1]) {
                arr[idx++] = arr[i];
            }
        }
        return idx;
    }
